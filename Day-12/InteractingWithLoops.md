blastoff <- function(x) {
    while (x != 0) {
        Sys.sleep(1)
        cat(x, "\n")
        x <- x - 1
    }
    Sys.sleep(1)
    cat("BANANAS!!!")
}

## Testing Human Response Times

```r
responseTime <- function(N) {
    readline("PRESS RETURN WHEN READY:")
    result <- rep(NA, N)
    interval <- c(0.1)
    for (k in 1:N) {
        before <- Sys.time()
        readline("PRESS RETURN:")
        after <- Sys.time()
        delay <- as.numeric(after - before)
        cat(rep("\n", 20))
        result[k] <- delay
        a <- Sys.time()
        Sys.sleep(rexp(20, rate = 1/2))
        b <- Sys.time()
        c <- as.numeric(b - a)
        interval <- c(interval, c)
    }
    latency <- result[1:N]
    finish <- data.frame(interval = interval[1:N], latency = latency)
    return(finish)
}
```
