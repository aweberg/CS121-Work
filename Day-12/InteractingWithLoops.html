        Sys.sleep(1)
        cat(x, "\n")
        x <- x - 1

responseTime <- function(N) {
    readline("PRESS RETURN WHEN READY:")
    result <- rep(NA, N)
    interval <- c(0.1)
    for (k in 1:N) {
        before <- Sys.time()
        readline("PRESS RETURN:")
        after <- Sys.time()
        delay <- as.numeric(after - before)
        cat(rep("\n", 20))
        result[k] <- delay
        a <- Sys.time()
        Sys.sleep(rexp(20, rate = 1/2))
        b <- Sys.time()
        c <- as.numeric(b - a)
        interval <- c(interval, c)
    }
    latency <- result[1:N]
    finish <- data.frame(interval = interval[1:N], latency = latency)
    return(finish)
}

testLatency2 <- function(N) {
    result <- vector(length = N)
    wordLength <- round(runif(N, 1, 20))
    for (k in 1:N) {
        x <- sample(letters, wordLength[k])
        word <- paste(x, collapse = "")  # Puts words together by amount of space between ''
        phrase <- paste("Type the word, '", word, "' ", sep = " ")
        command <- readline(phrase)

        Sys.sleep(runif(1, min = 1, max = 5))
        before <- Sys.time()

        while (command != word) {
            command <- readline("Look Again")
        }  # Waiting for it to look for word


        readline("Press Enter:")
        after <- Sys.time()
        delay <- as.numeric(after - before)
        cat(rep("\n", 20))
        result[k] <- delay

    }


    table <- data.frame(`Word Length` = WordLength, Latency = result)
    return(table)
}
